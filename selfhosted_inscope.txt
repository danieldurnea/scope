cat > main.py << 'EOF'
from fastapi import FastAPI, Request, Form, HTTPException
from fastapi.responses import HTMLResponse, RedirectResponse
from fastapi.templating import Jinja2Templates
import os
from dotenv import load_dotenv

load_dotenv()

app = FastAPI(title="LuciferвҖҷs Wrath C2")

templates = Jinja2Templates(directory="templates")

@router.get("/login", response_class=HTMLResponse)
async def login_page(request: Request):
    return templates.TemplateResponse("login.html", {"request": request, "step": 1})

@router.post("/login")
async def login_post(request: Request, username: str = Form(...), password: str = Form(...), totp_token: str = Form(None)):
    if username == "admin" and password == "admin123":
        if totp_token:
            request.session["user_id"] = 1
            return RedirectResponse(url="/dashboard", status_code=303)
        else:
            return templates.TemplateResponse("login.html", {"request": request, "step": 2})
    raise HTTPException(status_code=401, detail="Login eИҷuat")

@router.get("/dashboard", response_class=HTMLResponse)
async def dashboard(request: Request):
    if not request.session.get("user_id"):
        return RedirectResponse(url="/login")
    return templates.TemplateResponse("dashboard.html", {"request": request})
EOF
